\newcommand{\PolarisDivider}{%
  \begin{tikzpicture}[baseline=-0.55ex,x=1mm,y=1mm]
    \draw[
      draw=PolarisInk,
      line width=0.45pt,
      smooth,
      samples=121,
      domain=-48:48
    ] plot (\x,{-3.4*sin(180*\x/48)});

    \fill[PolarisInk]
      (0,4.1) -- (0.58,0.58) -- (4.1,0) -- (0.58,-0.58) --
      (0,-4.1) -- (-0.58,-0.58) -- (-4.1,0) -- (-0.58,0.58) -- cycle;
    \begin{scope}[rotate=45,scale=0.55]
      \fill[PolarisInk]
        (0,4.1) -- (0.58,0.58) -- (4.1,0) -- (0.58,-0.58) --
        (0,-4.1) -- (-0.58,-0.58) -- (-4.1,0) -- (-0.58,0.58) -- cycle;
    \end{scope}
  \end{tikzpicture}%
}
